def convert_temperature(value, from_unit, to_unit): if from_unit == 'celsius': if to_unit == 'fahrenheit': return (value * 1.8) + 32 elif to_unit == 'kelvin': return value + 273.15 elif to_unit == 'celsius': return value elif from_unit == 'fahrenheit': if to_unit == 'celsius': return (value - 32) / 1.8 elif to_unit == 'kelvin': return (value - 32) / 1.8 + 273.15 elif to_unit == 'fahrenheit': return value elif from_unit == 'kelvin': if to_unit == 'celsius': return value - 273.15 elif to_unit == 'fahrenheit': return (value - 273.15) * 1.8 + 32 elif to_unit == 'kelvin': return value def on_convert(event): try: value = float(Element('inputValue').element.value) from_unit = Element('from_unit').element.value to_unit = Element('to_unit').element.value converted_value = convert_temperature(value, from_unit, to_unit) Element('outputValue').element.value = f"{converted_value:.4f}" except ValueError: Element('outputValue').element.value = "Invalid input" Element('convertButton').element.onclick = on_convert
